docker-compose 命令 

###以下面yml文件为例

#查看容器状态相关
docker-compose ps -a

#查看容器日志
docker-compose logs -f h5st

#进入对应容器打开一个bash终端
docker-compose exec mix bash

#验证并显示docker-compose.yml文件的配置内容
docker-compose config

#拉取镜像
docker-compose pull h5st

#后台启动一个容器
docker-compose up -d h5st

#停止容器
docker-compose stop h5st

#删除容器
docker-compose rm h5st

#查看所有服务的镜像信息
docker-compose images

#删除镜像
docker rmi [镜像id]

#查看镜像摘要值-sha256
docker inspect [镜像id]
